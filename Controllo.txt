
//FUNZIONE-CONTROLLO
public int ScegliIlNumero(int S)
	{
		while(S == 0)
		{
			try
			{
				S = Console.getTastiera().nextInt();
			}
			catch(InputMismatchException e)
			{
				System.out.println();
				System.out.print("Valore inserito non valido" + "\n"+ "Scelta: ");
				Console.getTastiera().nextLine();
			}
		}
		return S;
	}

//NEI VARI SOTTOMENU

int scelta = 0;
scelta = ScegliIlNumero(scelta);
