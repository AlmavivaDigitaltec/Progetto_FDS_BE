
CREATE TABLE utente(
  MATRICOLA VARCHAR(4) PRIMARY KEY,
  NOME VARCHAR(20) NOT NULL,
  COGNOME VARCHAR(20) NOT NULL,
  MAIL   VARCHAR(30) NOT NULL,
  PASSWORD VARCHAR(20) NOT NULL
)

CREATE TABLE prenotabili(
  CODICE VARCHAR(5) PRIMARY KEY,
  NOME  VARCHAR(20) NOT NULL,
  DATA_INIZIO DATE NOT NULL,
  DATA_FINE DATE NOT NULL,
  ORARIO_INIZIO TIME NOT NULL,
  ORARIO_FINE  TIME NOT NULL
)

CREATE TABLE prenotazioni(
  ID_P   VARCHAR(5) PRIMARY KEY,
  MATRICOLA VARCHAR(4) REFERENCES UTENTE(MATRICOLA) ON DELETE CASCADE,
  CODICE  VARCHAR(5) REFERENCES PRENOTABILI(CODICE) ON DELETE CASCADE,
  DATA_PRENOTAZIONE DATE NOT NULL,
  ORA_INIZIO_P TIME NOT NULL,
  ORA_FINE_P TIME NOT NULL
)
